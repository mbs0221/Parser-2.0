struct Student {
    string name;
    int age;
    double score;
}

function getName(self) {
    return self.name;
}

function getAge(self) {
    return self.age;
}

function getScore(self) {
    return self.score;
}

function setScore(self, newScore) {
    self.score = newScore;
    return self.score;
}

let student1 = Student("Alice", 20, 95.5);
let student2 = Student("Bob", 22, 88.0);

print(student1.getName());
print(student1.getAge());
print(student1.getScore());

print(student2.getName());
print(student2.getAge());
print(student2.getScore());

print(student1.setScore(98.0));
print(student1.getScore());
