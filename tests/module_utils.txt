// 工具模块 - 提供实用函数

function max(a, b) {
    if (a > b) {
        return a;
    } else {
        return b;
    }
}

function min(a, b) {
    if (a < b) {
        return a;
    } else {
        return b;
    }
}

function abs(x) {
    if (x < 0) {
        return -x;
    } else {
        return x;
    }
}

function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

// 输出函数
function println(x) {
    print(x);
    print("\n");
}

function printInt(x) {
    print(x);
}

function printFloat(x) {
    print(x);
}

function printString(s) {
    print(s);
}

function printLine() {
    print("\n");
}

function printSeparator() {
    print("----------------------------------------\n");
}

// 格式化输出函数
function printResult(label, value) {
    print(label);
    print(" = ");
    print(value);
    print("\n");
}

function printHeader(title) {
    print("=== ");
    print(title);
    print(" ===\n");
}
