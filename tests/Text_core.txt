function add(x, y) {
    return x + y;
}

function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

let result1 = add(15, 25);
print("add(15, 25) = " + result1 + "\n");

let result2 = factorial(4);
print("factorial(4) = " + result2 + "\n");
