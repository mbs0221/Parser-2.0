print("Testing do-while loop:\n");

let i = 1;
do {
    print("i = ");
    print(i);
    print("\n");
    
    if (i == 2) {
        print("Breaking at i = 2\n");
        break;
    }
    
    i = i + 1;
} while (i <= 3);

print("After do-while loop\n\n");

// 测试continue
print("Testing do-while loop with continue:\n");

let j = 1;
do {
    if (j == 2) {
        print("Skipping j = 2\n");
        j = j + 1;
        continue;
    }
    
    print("j = ");
    print(j);
    print("\n");
    
    j = j + 1;
} while (j <= 3);

print("After do-while loop with continue\n");
