// 测试文件操作函数
function writeTestFile() {
    let handle = open("test_output.txt", "w");
    if (handle) {
        write(handle, "Hello, World!");
        write(handle, "This is a test file.", true);
        close(handle);
        println("File written successfully");
    } else {
        println("Failed to open file for writing");
    }
}

function readTestFile() {
    let readHandle = open("test_output.txt", "r");
    if (readHandle) {
        let line1 = read(readHandle);
        let line2 = read(readHandle);
        close(readHandle);
        
        println("Line 1: ", line1);
        println("Line 2: ", line2);
    } else {
        println("Failed to open file for reading");
    }
}

// 执行文件操作测试
writeTestFile();
readTestFile();
