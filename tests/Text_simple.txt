// 基本变量声明和赋值
let a = 10;
let b = 5;
let c = 0;

// 基本算术运算
c = a + b;
print("a + b = " + c + "\n");

c = a - b;
print("a - b = " + c + "\n");

c = a * b;
print("a * b = " + c + "\n");

c = a / b;
print("a / b = " + c + "\n");

c = a % b;
print("a % b = " + c + "\n");

// 数组操作
let arr = [1, 2, 3, 4, 5];
let arr_size = count(arr);
print("Array size: " + arr_size + "\n");

// 字符串操作
let str1 = "Hello";
let str2 = "World";
let combined = str1 + " " + str2;
print("Combined: " + combined + "\n");

// 逻辑运算
let x = 10;
let y = 5;
print("x > y: " + (x > y) + "\n");
print("x == y: " + (x == y) + "\n");
print("x != y: " + (x != y) + "\n");
print("x && y: " + (x && y) + "\n");
print("!x: " + (!x) + "\n");

// 条件语句
if (x > y) {
    print("x is greater than y\n");
} else {
    print("x is not greater than y\n");
}

// 循环
let i = 0;
while (i < 3) {
    print("i = " + i + "\n");
    i = i + 1;
}

// 函数定义和调用
function add(a, b) {
    return a + b;
}

function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

let result1 = add(10, 20);
print("add(10, 20) = " + result1 + "\n");

let result2 = factorial(5);
print("factorial(5) = " + result2 + "\n");
