// 测试复杂的类实例化和结构体实例化功能

// 定义类
class Student {
    public string name;
    public int age;
    public int grade;
}

// 定义结构体
struct Person {
    string name;
    int age;
}

// 定义另一个类
class Teacher {
    public string name;
    public string subject;
    public int experience;
}

// 测试类实例化
let student1 = Student("Alice", 20, 85);
let student2 = Student("Bob", 22, 90);

// 测试结构体实例化
let person1 = Person {name: "John", age: 25};
let person2 = Person {name: "Jane", age: 30};

// 测试另一个类实例化
let teacher1 = Teacher("Smith", "Math", 5);
