// 几何模块 - 提供几何计算函数

function circleArea(radius) {
    let pi = 3;
    return pi * radius * radius;
}

function rectangleArea(width, height) {
    return width * height;
}

function triangleArea(base, height) {
    return (base * height) / 2;
}

function distance(x1, y1, x2, y2) {
    let dx = x2 - x1;
    let dy = y2 - y1;
    // 简化处理，返回dx和dy的平方和
    return dx * dx + dy * dy;
}
