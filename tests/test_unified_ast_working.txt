// 测试统一的AST结构
// 验证结构体实例化现在使用IdentifierExpression

struct Person {
    string name;
    int age;
}

// 测试结构体实例化
let person1 = Person {name: "Alice", age: 25};

// 测试函数调用（类实例化本质上就是函数调用）
function createPerson(name, age) {
    return Person {name: name, age: age};
}

let person2 = createPerson("Bob", 30);

print("person1.name = ");
print(person1.name);
print("\n");

print("person2.age = ");
print(person2.age);
print("\n");
