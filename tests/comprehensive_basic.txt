// ==================== 基础功能综合测试 ====================

print("=== 基础功能综合测试 ===
");

// 1. 变量声明和基本运算
print("1. 变量声明和基本运算:");
let a = 10;
let b = 5;
print("a = " + a);
print("b = " + b);
let sum = a + b;
print("sum = " + sum);
print("
");

// 2. 字符串操作
print("2. 字符串操作:");
let str = "Hello World";
print("String: " + str);
print("Length: " + count(str));
let firstChar = str[0];
print("First character: " + firstChar);
print("
");

// 3. 数组操作
print("3. 数组操作:");
let arr = [1, 2, 3, 4, 5];
print("Array count: " + count(arr));
print("First element: " + arr[0]);
print("Third element: " + arr[2]);
print("
");

// 4. 条件语句
print("4. 条件语句:");
if (a > b) {
    print("a is greater than b");
} else {
    print("b is greater than or equal to a");
}

if (a == 10) {
    print("a equals 10");
}

if (b != 10) {
    print("b does not equal 10");
}
print("
");

// 5. 循环语句
print("5. 循环语句:");
let i = 0;
while (i < 3) {
    print("while loop i = " + i);
    i = i + 1;
}

for (let j = 0; j < 3; j = j + 1) {
    print("for loop j = " + j);
}
print("
");

// 6. 函数定义和调用
print("6. 函数定义和调用:");
function add(x, y) {
    return x + y;
}

function multiply(x, y) {
    return x * y;
}

let result1 = add(5, 3);
let result2 = multiply(4, 6);
print("add(5, 3) = " + result1);
print("multiply(4, 6) = " + result2);
print("
");

print("8. 布尔运算:");
