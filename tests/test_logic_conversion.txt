let a = 10;
let b = 5;

function println(x) {
    print(x + "\n");
}

function fib(n) {
    if (n <= 1) {
        return n;
    }
    return fib(n - 1) + fib(n - 2);
}

let n = 10;
let result6 = fib(n);
println("fib(" + n + ") = " + result6);

let result1 = a > b;
println("a > b = " + result1);

let result2 = a < b;
println("a < b = " + result2);

let result3 = a == b;
println("a == b = " + result3);

let result4 = a > 0 && b > 0;
println("a > 0 && b > 0 = " + result4);

let result5 = !(a > 0);
println("!(a > 0) = " + result5);
