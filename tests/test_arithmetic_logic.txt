// 算术逻辑运算测试

// 基本算术运算
let a = 10;
let b = 5;
let c = 3;

function println(x) {
    print(x + "\n");
}

println("=== 基本算术运算 ===");
println("a = " + a);
println("b = " + b);
println("c = " + c);

let sum = a + b;
println("a + b = " + sum);

let diff = a - b;
println("a - b = " + diff);

let product = a * b;
println("a * b = " + product);

let quotient = a / b;
println("a / b = " + quotient);

let remainder = a % c;
println("a % c = " + remainder);

// 比较运算
println("=== 比较运算 ===");
let eq = a == b;
println("a == b: " + eq);

let ne = a != b;
println("a != b: " + ne);

let lt = a < b;
println("a < b: " + lt);

let gt = a > b;
println("a > b: " + gt);

let le = a <= b;
println("a <= b: " + le);

let ge = a >= b;
println("a >= b: " + ge);

// 逻辑运算
println("=== 逻辑运算 ===");
let and_result = a > 0 && b > 0;
println("a > 0 && b > 0: " + and_result);

let or_result = a > 20 || b > 10;
println("a > 20 || b > 10: " + or_result);

let not_and = !(a > 0 && b > 0);
println("!(a > 0 && b > 0): " + not_and);

// 复杂表达式
println("=== 复杂表达式 ===");
let complex1 = (a + b) * c - (a / b);
println("(a + b) * c - (a / b) = " + complex1);

let complex2 = a > b && c < a || b == 5;
println("a > b && c < a || b == 5: " + complex2);

// 条件语句测试
println("=== 条件语句测试 ===");
if a > b then {
    println("a 大于 b");
} else {
    println("a 不大于 b");
}

if a == 10 && b == 5 then {
    println("a 等于 10 且 b 等于 5");
} else {
    println("条件不满足");
}

// 循环测试
println("=== 循环测试 ===");
let i = 0;
while i < 5 do {
    println("i = " + i);
    i = i + 1;
}

// 字符串和数字混合运算
println("=== 字符串和数字混合运算 ===");
let str = "Hello";
let num = 123;
let mixed = str + " " + num;
println("字符串拼接: " + mixed);

// 数组操作
println("=== 数组操作 ===");
let arr = [1, 2, 3, 4, 5];
println("数组长度: " + count(arr));
println("第一个元素: " + arr[0]);
println("最后一个元素: " + arr[4]);

// 字典操作
println("=== 字典操作 ===");
let person = {"name": "Alice", "age": 25};
println("姓名: " + person["name"]);
println("年龄: " + person["age"]);
