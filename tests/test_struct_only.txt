// 只测试结构体功能

// 定义结构体
struct Person {
    string name;
    int age;
}

// 测试结构体实例化
let person1 = Person {name: "John", age: 25};
print("Person: " + person1.name + ", Age: " + person1.age + "\n");

// 测试另一个结构体实例
let person2 = Person {name: "Alice", age: 30};
print("Person2: " + person2.name + ", Age: " + person2.age + "\n");

// 测试成员访问
let name = person1.name;
let age = person1.age;
print("Name: " + name + ", Age: " + age + "\n");
