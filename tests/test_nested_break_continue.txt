// 测试嵌套循环中的break和continue

let i = 0;
let j = 0;
let count = 0;

print("Testing nested loops with break and continue:\n");

while (i < 3) {
    i = i + 1;
    print("Outer loop i = ");
    print(i);
    print("\n");
    
    j = 0;
    while (j < 4) {
        j = j + 1;
        
        if (i == 2 && j == 2) {
            print("  Skipping i=");
            print(i);
            print(", j=");
            print(j);
            print("\n");
            continue;
        }
        
        if (i == 2 && j == 4) {
            print("  Breaking inner loop at i=");
            print(i);
            print(", j=");
            print(j);
            print("\n");
            break;
        }
        
        count = count + 1;
        print("  i=");
        print(i);
        print(", j=");
        print(j);
        print(", count=");
        print(count);
        print("\n");
    }
    
    if (i == 3) {
        print("Breaking outer loop at i=");
        print(i);
        print("\n");
        break;
    }
}

print("Final count = ");
print(count);
print("\n");
