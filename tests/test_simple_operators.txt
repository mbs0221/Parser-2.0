// 简单运算符测试

let a = 10;
let b = 5;

print("a = " . a);
print("b = " . b);

// 基本算术运算
let sum = a + b;
print("sum = " . sum);

let diff = a - b;
print("diff = " . diff);

// 比较运算
let gt = a > b;
print("gt = " . gt);

let lt = a < b;
print("lt = " . lt);

let eq = a == b;
print("eq = " . eq);

let ne = a != b;
print("ne = " . ne);

// 逻辑运算
let and_result = a > 0 && b > 0;
print("and_result = " . and_result);

let or_result = a > 20 || b > 10;
print("or_result = " . or_result);
