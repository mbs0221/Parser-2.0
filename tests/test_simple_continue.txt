let i = 1;
print("Testing simple continue:\n");

while (i <= 3) {
    print("i = ");
    print(i);
    print("\n");
    
    if (i == 2) {
        print("Continue at i = 2\n");
        i = i + 1;
        continue;
    }
    
    print("After continue check, i = ");
    print(i);
    print("\n");
    
    i = i + 1;
}

print("After loop, i = ");
print(i);
print("\n");
