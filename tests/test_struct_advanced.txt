// 复杂结构体测试

// 定义结构体
struct Person {
    string name;
    int age;
    string city;
}

struct Student {
    string name;
    int grade;
    int score;
}

// 测试结构体实例化
let person1 = Person {name: "John", age: 25, city: "New York"};
print("Person1: " + person1.name + ", Age: " + person1.age + ", City: " + person1.city + "\n");

let person2 = Person {name: "Alice", age: 30, city: "London"};
print("Person2: " + person2.name + ", Age: " + person2.age + ", City: " + person2.city + "\n");

let student1 = Student {name: "Bob", grade: 10, score: 95};
print("Student1: " + student1.name + ", Grade: " + student1.grade + ", Score: " + student1.score + "\n");

// 测试变量赋值
let name1 = person1.name;
let age1 = person1.age;
let city1 = person1.city;
print("Extracted - Name: " + name1 + ", Age: " + age1 + ", City: " + city1 + "\n");

// 测试字符串拼接
let info = "Student " + student1.name + " has score " + student1.score;
print(info + "\n");

// 测试条件语句中的结构体访问
if (student1.score > 90) {
    print("Excellent student: " + student1.name + "\n");
} else {
    print("Good student: " + student1.name + "\n");
}
