// 测试函数中的运算符重载

function abs(x) {
    if (x < 0) {
        return -x;
    } else {
        return x;
    }
}

function max(a, b) {
    if (a > b) {
        return a;
    } else {
        return b;
    }
}

function min(a, b) {
    if (a < b) {
        return a;
    } else {
        return b;
    }
}

function isPositive(x) {
    return x > 0;
}

function isNegative(x) {
    return x < 0;
}

function isZero(x) {
    return x == 0;
}

// 测试函数调用
let result1 = abs(-15);
print("abs(-15) = " + result1 + "\n");

let result2 = abs(20);
print("abs(20) = " + result2 + "\n");

let result3 = max(10, 25);
print("max(10, 25) = " + result3 + "\n");

let result4 = min(-5, 8);
print("min(-5, 8) = " + result4 + "\n");

let result5 = isPositive(7);
print("isPositive(7) = " + result5 + "\n");

let result6 = isNegative(-3);
print("isNegative(-3) = " + result6 + "\n");

let result7 = isZero(0);
print("isZero(0) = " + result7 + "\n");
