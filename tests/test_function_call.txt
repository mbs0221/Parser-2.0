function add(x, y) {
    return x + y;
}

function multiply(x, y) {
    return x * y;
}

function greet(name) {
    return "Hello, " + name;
}

let a = 10;
let b = 5;
let result1 = add(a, b);
print("add(" + a + ", " + b + ") = " + result1 + "\n");

let result2 = multiply(a, b);
print("multiply(" + a + ", " + b + ") = " + result2 + "\n");

let name = "World";
let greeting = greet(name);
print(greeting + "\n");
