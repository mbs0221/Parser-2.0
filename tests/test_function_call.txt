function add(x, y) {
    return x + y;
}

function multiply(x, y) {
    return x * y;
}

function greet(name) {
    return "Hello, " + name;
}

function println(str) {
    print(str + "\n");
}

let a = 10;
let b = 5;
let result1 = add(a, b);
println("add(" + a + ", " + b + ") = " + result1);

let result2 = multiply(a, b);
println("multiply(" + a + ", " + b + ") = " + result2);

let name = "World";
let greeting = greet(name);
println(greeting);
