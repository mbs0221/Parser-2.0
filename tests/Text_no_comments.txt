function add(a, b) {
    return a + b;
}

function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

let result1 = add(10, 20);
println("add(10, 20) = " + result1);

let result2 = factorial(5);
println("factorial(5) = " + result2);
