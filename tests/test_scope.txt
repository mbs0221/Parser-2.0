let outerVar = 100;
print("Outer variable: " + outerVar + "\n");

if (outerVar > 50) {
    let innerVar = 200;
    print("Inside if block - outerVar: " + outerVar + "\n");
    print("Inside if block - innerVar: " + innerVar + "\n");
    
    let temp = 300;
    print("Inside if block - temp: " + temp + "\n");
} else {
    let elseVar = 400;
    print("Inside else block - outerVar: " + outerVar + "\n");
    print("Inside else block - elseVar: " + elseVar + "\n");
}

print("After if-else - outerVar: " + outerVar + "\n");

let i = 0;
while (i < 4) {
    let loopVar = 500 + i;
    print("Inside while loop - i: " + i + "\n");
    print("Inside while loop - loopVar: " + loopVar + "\n");
    print("Inside while loop - outerVar: " + outerVar + "\n");
    i = i + 1;
}

print("After while loop - outerVar: " + outerVar + "\n");
print("After while loop - i: " + i + "\n");

function testScope() {
    let funcVar = 600;
    print("Inside function - funcVar: " + funcVar + "\n");
    print("Inside function - outerVar: " + outerVar + "\n");
    
    if (funcVar > 500) {
        let nestedVar = 700;
        print("Inside nested if - nestedVar: " + nestedVar + "\n");
        print("Inside nested if - funcVar: " + funcVar + "\n");
    }
    
    return funcVar;
}

let result = testScope();
print("Function result: " + result + "\n");
print("Final outerVar: " + outerVar + "\n");
