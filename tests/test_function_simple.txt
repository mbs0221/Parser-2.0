// 简单函数测试

function add(x, y) {
    return x + y;
}

function abs(x) {
    if (x < 0) {
        return -x;
    } else {
        return x;
    }
}

// 测试基本函数调用
let result1 = add(5, 3);
print("add(5, 3) = " + result1 + "\n");

let result2 = abs(-10);
print("abs(-10) = " + result2 + "\n");

let result3 = abs(5);
print("abs(5) = " + result3 + "\n");
