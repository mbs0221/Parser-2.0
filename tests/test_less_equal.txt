// 函数1：测试小于等于操作符
function testLessEqual() {
    println("小于等于测试开始");
    
    let i = 1;
    while (i <= 3) {
        println("循环次数");

    }
    
    println("小于等于测试完成");
}

    testLessEqual();
    
    println("小于等于操作符测试完成");
}
