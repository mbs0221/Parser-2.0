// 测试运算符重载

// 测试一元负号
let x = 10;
let y = -x;
print("x = " + x + ", y = " + y + "\n");

let z = -5;
print("z = " + z + "\n");

// 测试逻辑非
let a = 1;
let b = !a;
print("a = " + a + ", b = " + b + "\n");

let c = 0;
let d = !c;
print("c = " + c + ", d = " + d + "\n");

// 测试算术运算
let e = 15;
let f = 3;
print("e + f = " + (e + f) + "\n");
print("e - f = " + (e - f) + "\n");
print("e * f = " + (e * f) + "\n");
print("e / f = " + (e / f) + "\n");

// 测试比较运算
print("e > f = " + (e > f) + "\n");
print("e < f = " + (e < f) + "\n");
print("e == f = " + (e == f) + "\n");
print("e != f = " + (e != f) + "\n");

// 测试逻辑运算
print("e && f = " + (e && f) + "\n");
print("e || f = " + (e || f) + "\n");
print("!e = " + (!e) + "\n");
