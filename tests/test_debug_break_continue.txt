// 调试break和continue

let i = 1;
print("Testing break:\n");

while (i <= 3) {
    print("i = ");
    print(i);
    print("\n");
    
    if (i == 2) {
        print("Breaking at i = 2\n");
        break;
    }
    
    i = i + 1;
}

print("After break, i = ");
print(i);
print("\n\n");

// 测试continue
let j = 1;
print("Testing continue:\n");

while (j <= 3) {
    if (j == 2) {
        print("Skipping j = 2\n");
        j = j + 1;
        continue;
    }
    
    print("j = ");
    print(j);
    print("\n");
    
    j = j + 1;
}
