print("=== 单目运算符类型转换测试 ===");

print("1. 逻辑非运算:");
print("   !0 = ", !0);
print("   !1 = ", !1);
print("   !5 = ", !5);
print("   !0.0 = ", !0.0);
print("   !3.14 = ", !3.14);
print("   !true = ", !true);
print("   !false = ", !false);
print("   !\"\" = ", !"");
print("   !\"hello\" = ", !"hello");

print("2. 一元负号运算:");
print("   -5 = ", -5);
print("   -3.14 = ", -3.14);
print("   -0 = ", -0);
print("   -(-5) = ", -(-5));

print("3. 混合运算:");
print("   !(-5) = ", !(-5));
print("   -(!5) = ", -(!5));
print("   !(5 + 3) = ", !(5 + 3));
print("   -(5 + 3) = ", -(5 + 3));
