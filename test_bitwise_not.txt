print("=== 位运算取反测试 ===");

print("1. 整数位运算取反:");
print("   ~0 = ", ~0);
print("   ~1 = ", ~1);
print("   ~5 = ", ~5);
print("   ~255 = ", ~255);

print("2. 浮点数位运算取反:");
print("   ~3.14 = ", ~3.14);
print("   ~0.0 = ", ~0.0);

print("3. 混合运算:");
print("   ~(5 + 3) = ", ~(5 + 3));
print("   ~(-5) = ", ~(-5));
print("   ~(~5) = ", ~(~5));
