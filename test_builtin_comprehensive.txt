# 测试字符串函数
print("\n=== 字符串函数测试 ===\n");

let str = "Hello World";
print("length('Hello World') = ");
print(length(str));
print("\n");

print("substring('Hello World', 0, 5) = ");
print(substring(str, 0, 5));
print("\n");

print("upper('hello') = ");
print(upper("hello"));
print("\n");

print("lower('WORLD') = ");
print(lower("WORLD"));
print("\n");
