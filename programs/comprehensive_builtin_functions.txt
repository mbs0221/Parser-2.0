// ==================== 内置函数综合测试 ====================

println("=== 内置函数综合测试 ===
");

// 1. 数学函数
println("1. 数学函数:");
let x = -5;
let y = 10;
let z = 3.14;

println("abs(-5) = " + abs(x));
println("max(10, 5, 8) = " + max(y, 5, 8));
println("min(10, 5, 8) = " + min(y, 5, 8));
println("pow(2, 3) = " + pow(2, 3));
println("sqrt(16) = " + sqrt(16));
println("floor(3.7) = " + floor(3.7));
println("ceil(3.2) = " + ceil(3.2));
println("round(3.6) = " + round(3.6));
println("
");

// 2. 字符串函数
println("2. 字符串函数:");
let str = "Hello World";
println("length(\"Hello World\") = " + length(str));
println("substring(\"Hello World\", 0, 5) = " + substring(str, 0, 5));
println("upper(\"hello\") = " + upper("hello"));
println("lower(\"WORLD\") = " + lower("WORLD"));
println("trim(\"  hello  \") = " + trim("  hello  "));
println("indexOf(\"Hello World\", \"World\") = " + indexOf(str, "World"));
println("replace(\"Hello World\", \"World\", \"Universe\") = " + replace(str, "World", "Universe"));
println("
");

// 3. 类型转换函数
println("3. 类型转换函数:");
let num_str = "123";
let num_int = to_int(num_str);
println("to_int(\"123\") = " + num_int);

let double_str = "3.14";
let num_double = to_double(double_str);
println("to_double(\"3.14\") = " + num_double);

let str_result = to_string(42);
println("to_string(42) = " + str_result);

let bool_result = to_bool("true");
println("to_bool(\"true\") = " + bool_result);
println("
");

// 4. 随机数函数
println("4. 随机数函数:");
let rand1 = random();
println("random() = " + rand1);
let rand2 = random(100);
println("random(100) = " + rand2);
let rand3 = random(10, 20);
println("random(10, 20) = " + rand3);
println("
");

// 5. 数组函数
println("5. 数组函数:");
let arr = [1, 2, 3, 4, 5];
println("count([1,2,3,4,5]) = " + count(arr));
println("sum([1,2,3,4,5]) = " + sum(arr));
println("max([1,2,3,4,5]) = " + max(arr));
println("min([1,2,3,4,5]) = " + min(arr));
println("reverse([1,2,3,4,5]) = " + reverse(arr));
println("
");

// 6. 系统函数
println("6. 系统函数:");
println("time() = " + time());
println("date() = " + date());
println("version() = " + version());
println("
");

println("=== 内置函数测试完成 ===
");
