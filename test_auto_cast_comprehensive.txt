print("=== 全面的自动类型转换测试 ===");

print("1. 数值运算:");
print("   5 + 3.14 = ", 5 + 3.14);
print("   10 / 3 = ", 10 / 3);
print("   10 / 3.0 = ", 10 / 3.0);
print("   10 % 3 = ", 10 % 3);

print("2. 字符串运算:");
print("   \"Hello\" + 42 = ", "Hello" + 42);
print("   42 + \"Hello\" = ", 42 + "Hello");
print("   \"Result: \" + (5 + 3) = ", "Result: " + (5 + 3));

print("3. 比较运算:");
print("   5 > 3.14 = ", 5 > 3.14);
print("   3.14 < 5 = ", 3.14 < 5);
print("   10 == 10 = ", 10 == 10);
print("   10 != 5 = ", 10 != 5);
print("   5 >= 5 = ", 5 >= 5);
print("   3 <= 5 = ", 3 <= 5);

print("4. 布尔运算:");
print("   true && 5 = ", true && 5);
print("   false || 3 = ", false || 3);
print("   !0 = ", !0);
print("   !1 = ", !1);

print("5. 混合运算:");
print("   (5 + 3.14) * 2 = ", (5 + 3.14) * 2);
print("   (10 > 5) && (3 < 7) = ", (10 > 5) && (3 < 7));
print("   \"Sum: \" + (5 + 3) + \" is greater than 5: \" + ((5 + 3) > 5) = ", "Sum: " + (5 + 3) + " is greater than 5: " + ((5 + 3) > 5));
