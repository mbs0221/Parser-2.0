print("测试静态calculate方法");

// 测试Bool类型的静态calculate方法
let a = true;
let b = false;
print(a && b);  // 应该输出false
print(a || b);  // 应该输出true

// 测试Integer类型的静态calculate方法
let x = 10;
let y = 5;
print(x + y);   // 应该输出15
print(x - y);   // 应该输出5
print(x * y);   // 应该输出50
print(x / y);   // 应该输出2
print(x % y);   // 应该输出0

// 测试位运算
print(x & y);   // 应该输出0
print(x | y);   // 应该输出15
print(x ^ y);   // 应该输出15

// 测试比较运算
print(x > y);   // 应该输出true
print(x < y);   // 应该输出false
print(x == y);  // 应该输出false
print(x != y);  // 应该输出true

// 测试单目运算
print(!a);      // 应该输出false
print(-x);      // 应该输出-10
print(~x);      // 应该输出-11

print("测试完成");
