// 函数1：计算阶乘
function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

// 函数2：检查质数
function isPrime(num) {
    if (num < 2) {
        return false;
    }
    
    let i = 2;
    while (i * i <= num) {
        if (num % i == 0) {
            return false;
        }
        i = i + 1;
    }
    return true;
}

        print("计算完成");
        return result;
    } catch (error) {
        print("异常处理");
        return 0;
    }
}

// 函数4：算术运算测试
function testArithmetic(a, b, c) {
    let sum = a + b + c;
    let product = a * b * c;
    let difference = a - b - c;
    let quotient = a / b;
    let remainder = a % b;
    
    print("算术运算完成");
    print("和: " + sum);
    print("积: " + product);
    print("差: " + difference);
    print("商: " + quotient);
    print("余数: " + remainder);
    
    return sum + product;
}

// 函数5：逻辑运算测试
function testLogic(a, b, c) {
    let logicalAnd = a > 0 && b > 0 && c > 0;
    let logicalOr = a > 10 || b > 10 || c > 10;
    let logicalNot = !(a == 0);
    
    print("逻辑运算完成");
    print("逻辑与: " + logicalAnd);
    print("逻辑或: " + logicalOr);
    print("逻辑非: " + logicalNot);
    
    return logicalAnd;
}

// 函数6：比较运算测试
function testComparison(a, b) {
    let equal = a == b;
    let notEqual = a != b;
    let lessThan = a < b;
    let greaterThan = a > b;
    let lessEqual = a <= b;
    let greaterEqual = a >= b;
    
    print("比较运算完成");
    print("相等: " + equal);
    print("不等: " + notEqual);
    print("小于: " + lessThan);
    print("大于: " + greaterThan);
    print("小于等于: " + lessEqual);
    print("大于等于: " + greaterEqual);
    
    return equal;
}

// 函数7：位运算测试
function testBitwise(a, b) {
    let bitwiseAnd = a & b;
    let bitwiseOr = a | b;
    let bitwiseXor = a ^ b;
    let leftShift = a << 2;
    let rightShift = a >> 1;
    
    print("位运算完成");
    print("位与: " + bitwiseAnd);
    print("位或: " + bitwiseOr);
    print("位异或: " + bitwiseXor);
    print("左移: " + leftShift);
    print("右移: " + rightShift);
    
    return bitwiseAnd;
}

// 函数8：循环测试
function testLoops() {
    print("循环测试开始");
    
    // while循环
    let sum = 0;
    let j = 1;
    while (j <= 5) {
        sum = sum + j;
        j = j + 1;
    }
    print("1到5的和: " + sum);
    
    // 嵌套循环
    let product = 1;
    let k = 1;
    while (k <= 3) {
        let inner = 1;
        let l = 1;
        while (l <= k) {
            inner = inner * l;
            l = l + 1;
        }
        product = product * inner;
        k = k + 1;
    }
    print("嵌套循环结果: " + product);
}

// 函数9：分支测试
function testBranches() {
    print("分支测试开始");

    if (score >= 90) {
        print("优秀");
    } else if (score >= 80) {
        print("良好");
    } else if (score >= 70) {
        print("中等");
    } else if (score >= 60) {
        print("及格");
    } else {
        print("不及格");
    }
    
    // 嵌套if
    let age = 25;
    let income = 50000;
    
    if (age >= 18) {
        if (income >= 30000) {
            print("成年人，收入较高");
        } else {
            print("成年人，收入一般");
        }
    } else {
        print("未成年人");
    }
}

}
