let day = 3;
print("day = ", day);

function testSwitch() {
    switch (day) {
        case 1:
            print("Monday");
            break;
        case 2:
            print("Tuesday");
            break;
        case 3:
            print("Wednesday");
            break;
        case 4:
            print("Thursday");
            break;
        case 5:
            print("Friday");
            break;
        default:
            print("Weekend");
            break;
    }
}

testSwitch();
